\newcommand{\resumeSubheading}[5]{

% Resume item with two aligned pieces on the same line as the bullet
\newcommand{\resumeItemTwoPiece}[2]{%
  \item \begin{minipage}[t]{0.51\textwidth}\small #1\end{minipage}%
        \begin{minipage}[t]{0.40\textwidth}\small #2\end{minipage}%
  \vspace{-6pt}%
}

% Work entry with job, company, date split
\newcommand{\resumeItemTripleFirstBold}[3]{%
  \item
    \begin{minipage}[t]{0.55\textwidth}\small \textbf{#1}\end{minipage}%  % Job title
    \begin{minipage}[t]{0.30\textwidth}\small #2\end{minipage}%  % Company
    \hfill%
    \begin{minipage}[t]{0.12\textwidth}\raggedleft\small #3\end{minipage}%  % Date
  \vspace{-12pt}%
}

% Same layout as resumeItemTripleFirstBold, but larger font for key roles
\newcommand{\resumeItemTripleFirstBoldExtraBig}[3]{%
  \item
    \begin{minipage}[t]{0.55\textwidth}\normalsize \textbf{#1}\end{minipage}%  Job title
    \begin{minipage}[t]{0.30\textwidth}\normalsize #2\end{minipage}%          Company
    \hfill%
    \begin{minipage}[t]{0.12\textwidth}\raggedleft\normalsize #3\end{minipage}% Date
%  \vspace{-8pt}% adjust vertical space as you like
}

\newcommand{\resumeItemTripleSubtitle}[1]{
  \begin{minipage}[t]{0.85\textwidth}\small\textit{#1}\end{minipage}
  \vspace{-12pt}
}


\newcommand{\resumeItemTripleForBulletPoints}[3]{%
  \item
    \begin{minipage}[t]{0.51\textwidth}\small #1\end{minipage}% Col 1

    \hfill%
    \begin{minipage}[t]{0.12\textwidth}\raggedleft\small #3\end{minipage}% Col 3 - usually date
  \vspace{-6pt}%
}
